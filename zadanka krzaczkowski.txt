#include <stdio.h>
#include <wchar.h>

// wersja 1 ‚Äî dla char
void wyczysc(char tab[]) {
    tab[0] = '\0'; // pusty napis
}

// wersja 2 ‚Äî dla wchar_t
void wyczysc_w(wchar_t tab[]) {
    tab[0] = L'\0'; // pusty napis (szeroki znak)
}

int main() {
    char napis1[20] = "przykladowy tekst";
    wchar_t napis2[20] = L"drugi tekst";

    printf("Przed: \"%s\"\n", napis1);
    wyczysc(napis1);
    printf("Po:    \"%s\"\n\n", napis1);

    wprintf(L"Przed: \"%ls\"\n", napis2);
    wyczysc_w(napis2);
    wprintf(L"Po:    \"%ls\"\n", napis2);

    return 0;
}

// Funkcja do liczenia d≈Çugo≈õci napisu typu char
int dlugosc(char tekst[]) {
    int i = 0;
    while (tekst[i] != '\0') {
        i++;
    }
    return i;
}

    while (tekst[i] != L'\0') {
        i++;
    }
    return i;
}

// Funkcja por√≥wnujƒÖca dwa napisy leksykograficznie
int porzadek(char t1[], char t2[]) {
    int i = 0;
    while (t1[i] != '\0' && t2[i] != '\0') {
        if (t1[i] < t2[i]) return 1; // t1 jest wcze≈õniej
        if (t1[i] > t2[i]) return 0; // t2 jest wcze≈õniej
        i++;
    }

    // Je≈õli np. t1 = "ala", t2 = "alamakota" ‚Üí t1 kr√≥tszy = wcze≈õniejszy
    if (t1[i] == '\0' && t2[i] != '\0') return 1;

    // W przeciwnym przypadku ‚Äî t1 nie jest wcze≈õniej
    return 0;
}

    } else {
        printf("Pierwszy napis NIE jest wcze≈õniej leksykograficznie\n");
    }

    return 0;

// funkcja przepisujƒÖca tekst z z[] do do_[] dla char
void przepisz(char z[], char do_[]) {
    int i = 0;
    while (z[i] != '\0') {
        do_[i] = z[i];
        i++;
    }
    do_[i] = '\0'; // zako≈Ñczenie napisu
}

// funkcja przepisujƒÖca tekst dla wchar_t
void przepisz_w(wchar_t z[], wchar_t do_[]) {
    int i = 0;
    while (z[i] != L'\0') {
        do_[i] = z[i];
        i++;
    }
    do_[i] = L'\0';
}

    nap2[i] = L'\0';
}

void wytnijzn(char nap1[], char nap2[], char wynik[]) {

            }
            k++;
        }
        if (wystepuje) {
            wynik[j++] = nap1[i];
        }
        i++;
    }

    wynik[j] = '\0';

float odleglosc(struct punkt a, struct punkt b) {
    return sqrtf((a.x - b.x)*(a.x - b.x) +
                 (a.y - b.y)*(a.y - b.y) +
                 (a.z - b.z)*(a.z - b.z));
}

    float min = odleglosc(tab[0], tab[1]);

        for (int j = i + 1; j < n; j++) {
            float d = odleglosc(tab[i], tab[j]);
            if (d < min)
                min = d;
        }
    return min;
}

    struct zespolone wynik;

    // wynik ma byƒá typu float je≈õli kt√≥rykolwiek sk≈Çadnik jest float
    if (z1.tp == 1 || z2.tp == 1) {
        wynik.tp = 1;
        wynik.re.wymierna = 
            (z1.tp == 1 ? z1.re.wymierna : (float)z1.re.calkowita) + 
            (z2.tp == 1 ? z2.re.wymierna : (float)z2.re.calkowita);
        wynik.im.wymierna = 
            (z1.tp == 1 ? z1.im.wymierna : (float)z1.im.calkowita) +

        struct { float a, h; } trojkat;
        struct { float a, b; } prostokat;
        struct { float a, h; } rownoleglobok;

        case 3: return ((f.dane.trapez.a + f.dane.trapez.b) * f.dane.trapez.h) / 2;
        default: return 0;
    }
}

                prev->next = curr->next;
                free(curr);
                return Lista;
            }
        }
        prev = curr;
        curr = curr->next;
    }

    return Lista;
}
